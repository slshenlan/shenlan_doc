class QueueError extends Error {
  constructor(params) {
    super(params);
  }
}

class Queue {
  constructor() {
    this.data = [];
  }

  // 入队
  joinQueue(data) {
    this.data.push(...data);
  }

  // 出队
  outQueue() {
    // console.log(this.data);

    if (this.isEmpty()) {
      throw new QueueError('可读取数据列表为空');
    } else {
      return this.data.shift();
    }
  }

  // 查看队首元素
  getFirstItem() {
    if (this.isEmpty()) {
      return null;
    } else {
      return this.data[0];
    }
  }

  // 检查队列是否为空
  isEmpty() {
    return this.data.length === 0;
  }

  // 获取队列的大小
  size() {
    return this.data.length;
  }

  // 清空队列
  clear() {
    this.data = [];
  }
}

// 串口调试
export default class SlSerialPort {
  constructor(params) {
    const { contrastParams, handlePack, onConnect, onDisconnect, onFail } = params;

    // INFO: 该字段 根据 对接文档 具体配置
    this.contrastParams = contrastParams || {};
    this.handlePack = handlePack || function () {};
    this.onConnect = onConnect || function () {};
    this.onDisconnect = onDisconnect || function () {};
    this.onFail = onFail || function () {};

    // 初始化 队列对象
    this.queue = new Queue();

    // 处理后的数据队列
    this.handleQueue = new Queue();

    // 串口发送数据对象
    this.serialPortWriter = null;
  }
  // 串口连接
  async connect() {
    try {
      // 浏览器支持serial
      if ('serial' in navigator) {
        console.log('当前浏览器支持serial');
        // 提示用户选择一个串口
        const port = await navigator.serial.requestPort();
        // 获取用户之前授予网站访问权限的所有串口.
        // const ports = await navigator.serial.getPorts();
        //打开串口
        //baudRate：波特率
        //dataBits：数据位（7 或 8）
        //stopBits：停止位（1 或 2）
        //parity：校验位"none"、"偶even"、"奇odd"
        //bufferSize：读写缓冲区大小（必须小于 16MB）
        //flowControl：流量控制模式。"none"、"hardware"

        await port.open({ baudRate: 115200 });

        const reader = port.readable.getReader();
        this.serialPortWriter = port.writable.getWriter();

        // 成功连接
        this.onConnect({
          label: '连接成功'
        });

        try {
          // eslint-disable-next-line no-constant-condition
          while (true) {
            const { value, done } = await reader.read();

            // console.log(value);
            // console.log(value[0]);
            // console.log(decimalToHex(value[0]));

            if (done) {
              reader.releaseLock();
              break;
            }
            const arr = Array.from(value);
            // console.log(arr);

            this.queue.joinQueue(arr);

            try {
              this._callback(this.queue);
            } catch (error) {
              console.error(error);
            }
          }
        } catch (error) {
          console.error('读取发生错误');
          console.error(error);
        } finally {
          // 断开连接
          this.onDisconnect({ label: '断开连接' });
          //   try {
          //     reader.releaseLock();
          //   } catch (error) {
          //     console.log(error);
          //   }

          console.log('允许稍后关闭串口。');

          try {
            await port.close();
          } catch (error) {
            console.log(error);
          }
        }
      } else {
        console.log('当前浏览器不支持serial');
      }
    } catch (error) {
      // TODO: 连接发生错误
      console.error(`连接发生错误:${error}`);
      this.onFail({ label: '连接失败' });
    }
  }
  // 发送消息
  async send(value) {
    await this.serialPortWriter.write(new Uint8Array(value));
  }

  _callback(queue) {
    while (!queue.isEmpty()) {
      const bak = [];

      // TODO: 这里根据业务配置字段
      const { CRPC_300SDKPACKAGEHEADFIRST } = this.contrastParams;

      try {
        if (queue.getFirstItem() === CRPC_300SDKPACKAGEHEADFIRST) {
          //如果是包头，则忽略之前存的handleQueue包，直接解析新包
          this.handleQueue.clear();

          this.handlePack(queue, bak);
        } else {
          //如果不是包头，则循环将字节加到之前的handleQueue包尾，直到遇到包头或队列为空
          while (!queue.isEmpty()) {
            if (queue.getFirstItem() === CRPC_300SDKPACKAGEHEADFIRST) {
              break;
            }
            this.handleQueue.joinQueue([queue.outQueue()]);
          }
          //将已拼接好的handleQueue执行解析，同时将bak传进去，如果拼接好后还是不够，触发无可读数据报错，在catch中会将bak再加入handleQueue，等待下次读取
          this.handlePack(this.handleQueue, bak);
          //不报错的话，解析完之后handleQueue就可以清了，因为无论解析是否成功，剩余部分都没有包头了
          this.handleQueue.clear();
        }
      } catch (error) {
        if (error instanceof QueueError) {
          // console.log('主动抛出异常');

          //由于包不全导致读取时主动抛出异常，在此处接收，将读取到的临时bak加入handleQueue中
          this.handleQueue.clear();
          this.handleQueue.joinQueue(bak);

          //然后没有可读数据会不满足while条件跳出循环，去到串口数据读取位置，重新开始下一轮读取
        } else {
          console.log(error);

          //其他异常不关心，直接抛出
          throw error;
        }
      }
    }
  }
}

import { ref } from 'vue';
import usePc300Config from './pc300-config';
import usePc300Analyzer from './pc300-analyzer';
import SlSerialPort from './sl-serial-port';

export default function useSerialPortDevTool(onMessage) {
  const { packConfig } = usePc300Config();

  const { CRPC_300SDKPACKAGEHEADFIRST } = packConfig.token;

  const { serialPorAnalyzer } = usePc300Analyzer();

  // 串口连接状态
  const serialPortStatus = ref({
    status: '', // 连接状态
    text: '', //
    data: {
      success: {
        value: 0,
        label: '连接成功'
      },
      fail: {
        value: 1,
        label: '连接失败'
      },
      disconnect: {
        value: 2,
        label: '断开连接'
      }
    }
  });

  const serialPort = new SlSerialPort({
    contrastParams: {
      CRPC_300SDKPACKAGEHEADFIRST
    },
    handlePack: (queue, bak) => {
      // 处理串口包数据
      serialPorAnalyzer(queue, bak);
    },
    onConnect: () => {
      handleSerialPortConnected(serialPortStatus.value.data.success.value);
    },
    onDisconnect: () => {
      handleSerialPortConnected(serialPortStatus.value.data.disconnect.value);
    },
    onFail: () => {
      handleSerialPortConnected(serialPortStatus.value.data.fail.value);
    }
  });

  // 连接串口
  function serialPortInit() {
    serialPort.connect();
  }

  //  发送消息
  function serialPortSendMsg(value) {
    console.log('发送消息:', value);

    serialPort.send(value);
  }

  // 处理连接状态
  function handleSerialPortConnected(value) {
    const data = serialPortStatus.value.data;

    serialPortStatus.value.status = value;

    for (const key in data) {
      if (Object.hasOwnProperty.call(data, key)) {
        const element = data[key];

        if (element.value === value) {
          serialPortStatus.value.text = element.label;
        }
      }
    }

    console.log(serialPortStatus.value);
  }

  // 处理设备信息
  function handleDeviceInfo(params) {}

  return { serialPortStatus, serialPortInit, serialPortSendMsg };
}

//string转uint8array
function toUint8Arr(str) {
  const buffer = [];
  for (let i of str) {
    const _code = i.charCodeAt(0);
    if (_code < 0x80) {
      buffer.push(_code);
    } else if (_code < 0x800) {
      buffer.push(0xc0 + (_code >> 6));
      buffer.push(0x80 + (_code & 0x3f));
    } else if (_code < 0x10000) {
      buffer.push(0xe0 + (_code >> 12));
      buffer.push(0x80 + ((_code >> 6) & 0x3f));
      buffer.push(0x80 + (_code & 0x3f));
    }
  }
  return Uint8Array.from(buffer);
}
//uint8array转String
function uint8ArrayToString(serialData) {
  var out, i, len, c;
  var char2, char3;
  out = '';
  len = serialData.length;
  i = 0;
  while (i < len) {
    c = serialData[i++];
    switch (c >> 4) {
      case 0:
      case 1:
      case 2:
      case 3:
      case 4:
      case 5:
      case 6:
      case 7:
        // 0xxxxxxx
        out += String.fromCharCode(c);
        break;
      case 12:
      case 13:
        // 110x xxxx   10xx xxxx
        char2 = serialData[i++];
        out += String.fromCharCode(((c & 0x1f) << 6) | (char2 & 0x3f));
        break;
      case 14:
        // 1110 xxxx  10xx xxxx  10xx xxxx
        char2 = serialData[i++];
        char3 = serialData[i++];
        out += String.fromCharCode(
          ((c & 0x0f) << 12) | ((char2 & 0x3f) << 6) | ((char3 & 0x3f) << 0)
        );
        break;
    }
  }
  return out;
}

// 软件版本 Byte0:( a -- bit7-bit4 b –bit3-bit0)     (Ver a.b.c.d)
// 软件版本 Byte1:( c -- bit7-bit4 d –bit3-bit0)
// 硬件版本 Byte0:(Ver x.y x -- bit7-bit4 y –bit3-bit0)
// 电量等级: bit2-0  分为:0. 1. 2. 3 等级

const arr = [1, 2, 3, 4];

const v1 = arr[0];
const v2 = arr[1];
const v3 = arr[2];
const v4 = arr[3];

const deviceInfo = {

  hard: `${v3 >> 4}.${v3 & 0x0f}`,
  electricity: `${v4 & 0x07}`

// Byte1-2:SYS(收缩压，2 字节)，，高字节在前，低字节在后，其 中最高位(即 bit16)表示心率结果，” 0”心率正常，” 1”心率不齐。

const arr = [0，128];

const sys1 = arr[0];
const sys2 = arr[1];

// 结果
const sysStatus = (sys1 >> 7) & 0x01;

// 值
const sysValue = ((sys1 & 0x7F) << 8) | sys2;

// 示例三 (体温)

// Byte1:体温结果状态字，编码如下:
//       bit7: ” 0”手动启动测量(默认)，” 1”非手动启动测量
//       bit6-bit5: 状态
//                  00:待机状态(开机默认)
//                  01:正在测量状态(跟踪)
//                  10:测量结束
//                  11:测量超时(暂定 1 分钟未出最终结果)
// bit4: ” 0”测量中探头插入，” 1”测量中探头脱落
// bit3-bit0:保留

// Byte2:体温值高字节
// Byte3:体温值低字节

// 注:体温结果范围十进制 0-1301(正常范围十进制 200-1301)，体温值计算公式如下:
